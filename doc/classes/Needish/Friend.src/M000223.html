      def new_from_xml(xml)
       
        result = []

        xml.search("user").each do  |user|
          Friend.new do |u|
                u.id                =        user[:id] 
                u.name              =         user[:name]
                u.lastname          =         user[:lastname]
                u.nickname          =         user[:nickname]
                u.pictureurl        =         user[:pictureurl]
                u.displayname       =         user[:displayname]
           result << u
           end
           
        end

        return result
       
        
      end
